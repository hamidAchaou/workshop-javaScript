Enoncé

Créez 4 fonctions Javascript (Somme, Soustraction, Division, Multiplication). L’utilisateur saisie deux nombres dans deux inputs, choisi ensuite une opération arithmétique. Le résultat est donné par l’une des 4 fonctions Javascript.

Code source
                        function add() {
                            let inputOne = document.getElementById("inp-1").value;
                            let inputTwo = document.getElementById("inp-2").value;
                            let resultCalc = document.getElementById("resultCalc");
                        
                            resultCalc.value = Number(inputOne) + Number(inputTwo);
                            document.getElementById("pre1").style.visibility= "visible";
                        }
                        function subs() {
                            let inputOne = document.getElementById("inp-1").value;

                            let resultCalc = document.getElementById("resultCalc");
                        
                            resultCalc.value = Number(inputOne) - Number(inputTwo);
                            document.getElementById("pre1").style.visibility= "visible";
                        }
                        function multi() {
